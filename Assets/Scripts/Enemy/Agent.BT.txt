tree("Root")
    fallback
        tree("Priorities")
        tree("Attack")
        tree("Lay Trap")
        tree("Push")
        tree("Patrol")

tree("Priorities")
    fallback
        while IsDead
            Die
        while IsStunned
            Stun
        while IsTrapTriggered
            MoveToTrap

tree("Attack")
    sequence
        while IsAlerted
            Alert
        Prowl
        fallback
            while IsWithinAttackRange
                Attack
            sequence
                MoveToHidingLocation
                Hide
        Flee

tree("Lay Trap")
    sequence
        CanLayTrap
        IsAtCorridor
        LayTrap

tree("Push")
    sequence
        CanPush
        FindNearestWaitLocation
        MoveToWaitLocation
        Wait
        Push

tree("Patrol")
    Patrol
